#### 创建生成器逻辑 (backend/utils/generator.js)
```javascript
const fs = require('fs');
const path = require('path');

class PromptGenerator {
  constructor() {
    this.templates = this.loadTemplates();
  }
  
  loadTemplates() {
    const templatePath = path.join(__dirname, '../../data/templates.json');
    return JSON.parse(fs.readFileSync(templatePath, 'utf8'));
  }
  
  generate(input) {
    const results = [];
    
    for (const [styleKey, styleData] of Object.entries(this.templates)) {
      const template = this.selectRandomTemplate(styleData.templates);
      const prompt = template.replace('{input}', input);
      
      results.push({
        style: styleKey,
        styleName: styleData.name,
        prompt: prompt
      });
    }
    
    return results;
  }
  
  selectRandomTemplate(templates) {
    return templates[Math.floor(Math.random() * templates.length)];
  }
}

module.exports = PromptGenerator;
```

#### 创建API路由 (backend/routes/prompts.js)
```javascript
const express = require('express');
const PromptGenerator = require('../utils/generator');
const router = express.Router();

const generator = new PromptGenerator();

// 输入验证中间件
function validateInput(req, res, next) {
  const { input } = req.body;
  
  if (!input || typeof input !== 'string') {
    return res.status(400).json({
      success: false,
      error: {
        code: 'INVALID_INPUT',
        message: '输入内容无效'
      }
    });
  }
  
  const trimmedInput = input.trim();
  if (trimmedInput.length === 0) {
    return res.status(400).json({
      success: false,
      error: {
        code: 'EMPTY_INPUT',
        message: '输入内容不能为空'
      }
    });
  }
  
  if (trimmedInput.length > 500) {
    return res.status(400).json({
      success: false,
      error: {
        code: 'INPUT_TOO_LONG',
        message: '输入内容过长，请控制在500字符以内'
      }
    });
  }
  
  req.body.input = trimmedInput;
  next();
}

// 生成提示词
router.post('/generate', validateInput, (req, res) => {
  try {
    const { input } = req.body;
    const results = generator.generate(input);
    
    res.json({
      success: true,
      data: results
    });
  } catch (error) {
    console.error('生成提示词失败:', error);
    res.status(500).json({
      success: false,
      error: {
        code: 'GENERATION_FAILED',
        message: '生成失败，请重试'
      }
    });
  }
});

module.exports = router;
```

#### JavaScript功能 (frontend/script.js)
```javascript
// PromptAtelier 主应用
class PromptAtelier {
    constructor() {
        this.elements = this.initElements();
        this.currentInput = '';
        this.currentResults = [];
        this.initEventListeners();
    }
    
    initElements() {
        return {
            input: document.getElementById('prompt-input'),
            charCount: document.getElementById('char-count'),
            generateBtn: document.getElementById('generate-btn'),
            clearBtn: document.getElementById('clear-btn'),
            resultsSection: document.getElementById('results-section'),
            resultsContainer: document.getElementById('results-container'),
            copyAllBtn: document.getElementById('copy-all-btn'),
            regenerateBtn: document.getElementById('regenerate-btn'),
            toastContainer: document.getElementById('toast-container')
        };
    }
    
    initEventListeners() {
        // 输入框事件
        this.elements.input.addEventListener('input', (e) => {
            this.updateCharCount();
            this.validateInput();
        });
        
        // 按钮事件
        this.elements.generateBtn.addEventListener('click', () => this.generatePrompts());
        this.elements.clearBtn.addEventListener('click', () => this.clearInput());
        this.elements.copyAllBtn.addEventListener('click', () => this.copyAllResults());
        this.elements.regenerateBtn.addEventListener('click', () => this.generatePrompts());
        
        // 键盘事件
        this.elements.input.addEventListener('keydown', (e) => {
            if (e.ctrlKey && e.key === 'Enter') {
                this.generatePrompts();
            }
        });
    }
    
    updateCharCount() {
        const count = this.elements.input.value.length;
        this.elements.charCount.textContent = count;
        
        if (count > 450) {
            this.elements.charCount.className = 'text-red-500 font-bold';
        } else if (count > 350) {
            this.elements.charCount.className = 'text-yellow-500';
        } else {
            this.elements.charCount.className = 'text-gray-400';
        }
    }
    
    validateInput() {
        const input = this.elements.input.value.trim();
        const isValid = input.length > 0 && input.length <= 500;
        this.elements.generateBtn.disabled = !isValid;
    }
    
    clearInput() {
        this.elements.input.value = '';
        this.updateCharCount();
        this.validateInput();
        this.elements.input.focus();
        this.hideResults();
    }
    
    async generatePrompts() {
        const input = this.elements.input.value.trim();
        if (!input) {
            this.showToast('请输入内容', 'error');
            return;
        }
        
        this.setLoading(true);
        
        try {
            const response = await fetch('/api/generate', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json'
                },
                body: JSON.stringify({ input })
            });
            
            const result = await response.json();
            
            if (result.success) {
                this.currentInput = input;
                this.currentResults = result.data;
                this.displayResults(result.data);
                this.showToast('生成成功！', 'success');
            } else {
                this.showToast(result.error.message || '生成失败', 'error');
            }
        } catch (error) {
            console.error('API调用失败:', error);
            this.showToast('网络错误，请重试', 'error');
        } finally {
            this.setLoading(false);
        }
    }
    
    setLoading(loading) {
        const generateText = this.elements.generateBtn.querySelector('.generate-text');
        const loadingText = this.elements.generateBtn.querySelector('.loading-text');
        
        if (loading) {
            generateText.classList.add('hidden');
            loadingText.classList.remove('hidden');
            this.elements.generateBtn.disabled = true;
        } else {
            generateText.classList.remove('hidden');
            loadingText.classList.add('hidden');
            this.elements.generateBtn.disabled = false;
        }
    }
    
    displayResults(results) {
        this.elements.resultsContainer.innerHTML = '';
        
        results.forEach((result, index) => {
            const card = this.createResultCard(result, index);
            this.elements.resultsContainer.appendChild(card);
        });
        
        this.elements.resultsSection.classList.remove('hidden');
        this.elements.resultsSection.scrollIntoView({ behavior: 'smooth' });
    }
    
    createResultCard(result, index) {
        const card = document.createElement('div');
        card.className = 'result-card bg-white rounded-lg shadow-md p-6 border border-gray-200';
        
        const styleNames = {
            professional: '专业逻辑型',
            creative: '创意激发型',
            simple: '简洁实用型'
        };
        
        const styleColors = {
            professional: 'bg-blue-100 text-blue-800',
            creative: 'bg-purple-100 text-purple-800',
            simple: 'bg-green-100 text-green-800'
        };
        
        card.innerHTML = `
            <div class="flex justify-between items-start mb-3">
                <span class="inline-block px-3 py-1 rounded-full text-sm font-medium ${styleColors[result.style]}">
                    ${styleNames[result.style]}
                </span>
                <button
                    class="copy-btn text-gray-400 hover:text-gray-600 focus:outline-none"
                    data-index="${index}"
                    title="复制这条提示词"
                >
                    📋
                </button>
            </div>
            <p class="text-gray-700 leading-relaxed">${result.prompt}</p>
        `;
        
        // 添加复制事件
        const copyBtn = card.querySelector('.copy-btn');
        copyBtn.addEventListener('click', () => this.copyResult(index));
        
        return card;
    }
    
    async copyResult(index) {
        const result = this.currentResults[index];
        if (!result) return;
        
        try {
            await navigator.clipboard.writeText(result.prompt);
            this.showToast('已复制到剪贴板', 'success');
        } catch (error) {
            // 降级方案
            this.fallbackCopyToClipboard(result.prompt);
        }
    }
    
    async copyAllResults() {
        if (this.currentResults.length === 0) return;
        
        const allPrompts = this.currentResults.map((result, index) => {
            const styleNames = {
                professional: '专业逻辑型',
                creative: '创意激发型',
                simple: '简洁实用型'
            };
            return `${index + 1}. ${styleNames[result.style]}:\n${result.prompt}`;
        }).join('\n\n');
        
        const fullText = `PromptAtelier 生成结果\n输入内容：${this.currentInput}\n\n${allPrompts}`;
        
        try {
            await navigator.clipboard.writeText(fullText);
            this.showToast('全部内容已复制', 'success');
        } catch (error) {
            this.fallbackCopyToClipboard(fullText);
        }
    }
    
    fallbackCopyToClipboard(text) {
        const textArea = document.createElement('textarea');
        textArea.value = text;
        document.body.appendChild(textArea);
        textArea.select();
        
        try {
            document.execCommand('copy');
            this.showToast('已复制到剪贴板', 'success');
        } catch (error) {
            this.showToast('复制失败，请手动复制', 'error');
        } finally {
            document.body.removeChild(textArea);
        }
    }
    
    hideResults() {
        this.elements.resultsSection.classList.add('hidden');
        this.currentResults = [];
    }
    
    showToast(message, type = 'info') {
        const toast = document.createElement('div');
        const typeColors = {
            success: 'bg-green-500',
            error: 'bg-red-500',
            info: 'bg-blue-500'
        };
        
        toast.className = `toast ${typeColors[type]} text-white px-4 py-2 rounded-md shadow-lg mb-2`;
        toast.textContent = message;
        
        this.elements.toastContainer.appendChild(toast);
        
        setTimeout(() => {
            toast.classList.add('hide');
            setTimeout(() => {
                if (toast.parentNode) {
                    toast.parentNode.removeChild(toast);
                }
            }, 300);
        }, 3000);
    }
}

// 初始化应用

});
```
