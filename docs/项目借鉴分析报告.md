### 2.2 核心特性分析

#### ✅ **结构化提示词构建器 (Prompt Builder)**

这是该项目最有价值的功能之一：

```typescript
// 流式 API 构建提示词
const prompt = builder()
  .role("Senior TypeScript Developer")
  .context("You are helping review code")
  .task("Analyze the following code for bugs")
  .constraints(["Be concise", "Focus on critical issues"])
  .output("JSON with { bugs: [], suggestions: [] }")
  .variable("code", { required: true })
  .build();
```

**核心方法**:
- `.role()` - 设置角色
- `.context()` - 设置上下文
- `.task()` - 设置任务
- `.constraints()` - 设置约束条件
- `.output()` - 设置输出格式
- `.variable()` - 定义变量
- `.fewShot()` - 添加示例(Few-shot)
- `.example()` - 添加输入输出示例

#### ✅ **多媒体提示词构建器**

支持图像、视频、音频提示词的专业化构建：


#### 5.1.1 引入流式提示词构建器

**目标**: 实现链式调用的提示词构建API

**新建文件**: `backend/utils/promptBuilder.js`

```javascript
class PromptBuilder {
  constructor() {
    this.data = {
      role: '',
      context: '',
      task: '',
      constraints: [],
      output: '',
      variables: [],
      examples: []
    };
  }

  role(r) { this.data.role = r; return this; }
  context(c) { this.data.context = c; return this; }
  task(t) { this.data.task = t; return this; }
  constraints(arr) { this.data.constraints = arr; return this; }
  output(o) { this.data.output = o; return this; }
  
  variable(name, options = {}) {
    this.data.variables.push({ name, ...options });
    return this;
  }
  
  example(input, output) {
    this.data.examples.push({ input, output });
    return this;
  }
  
  build() {
    let prompt = '';
    if (this.data.role) prompt += `你是一位${this.data.role}。\n\n`;
    if (this.data.context) prompt += `背景: ${this.data.context}\n\n`;
    if (this.data.task) prompt += `任务: ${this.data.task}\n\n`;
    if (this.data.constraints.length) {
      prompt += '约束条件:\n';
      this.data.constraints.forEach(c => prompt += `- ${c}\n`);
      prompt += '\n';
    }
    if (this.data.examples.length) {
      prompt += '示例:\n';
      this.data.examples.forEach(e => {
        prompt += `输入: ${e.input}\n输出: ${e.output}\n\n`;
      });
    }
    if (this.data.output) prompt += `输出格式: ${this.data.output}\n`;
    
    return {
      content: prompt,
      variables: this.data.variables,
      metadata: this.data
    };
  }
  
  toJSON() { return JSON.stringify(this.build(), null, 2); }
  toYAML() { /* YAML转换 */ }
}

module.exports = { PromptBuilder, builder: () => new PromptBuilder() };
```

```javascript
// 变量解析器
class VariableParser {
  static parse(template) {
    const regex = /\$\{(\w+)(?::([^}]*))?\}/g;
    const variables = [];
    let match;
    
    while ((match = regex.exec(template)) !== null) {
      variables.push({
        name: match[1],
        defaultValue: match[2] || '',
        fullMatch: match[0]
      });
    }
    
    return variables;
  }
  
  static interpolate(template, values) {
    return template.replace(/\$\{(\w+)(?::([^}]*))?\}/g, (match, name, def) => {
      return values[name] !== undefined ? values[name] : (def || `[${name}]`);
    });
  }
}
```
